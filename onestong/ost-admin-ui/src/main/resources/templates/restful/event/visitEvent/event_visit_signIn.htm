json 参数示例

              {
    "event": {
        "userId": "3",                			------------用户 id
        "veId": "sfsfsf-sfsfsf-sfsf111111",		------------事件 uuid
        "vistedAddress": "湖北武汉",				------------签到地点
        "vistedLongitude": "112.2323",			------------签到经度
        "visitedLatitude": "23.34",				------------签到纬度
        "title": "我去韦德办事",					------------外访事件title
        "content": "testtest"					------------外访事件描述（说点什么呢）
    },
    "pTag": {                   				------------
        "pTId": "2",							------------项目标签 id（新增标签 没有此项）
        "pTName": "韦德"
    },
    "cTags": [									------------客户标签（数组） 
        {
            "cTId": "11",						------------客户标签 id（新增标签 没有此项）
            "cTName": "进行中"
        },
        {
            "cTId": "12"
        }
    ]
}
